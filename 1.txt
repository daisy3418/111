def execute_intent(user_text):
    lines = []

    try:
        try:
            rules = agent_parse_intent(user_text)
            parser_name = "LangChain Intent Agent"
        except Exception as llm_error:
            rules = parse_intent(user_text)
            parser_name = "本地规则解析器"

        lines.append(f"### 🤖 解析方式\n{parser_name}\n")

        lines.append("### 1. AI 解析出的结构化策略")
        for rule in rules:
            lines.append(f"- `{rule['src']} -> {rule['dst']} : {rule['action']}`")
        lines.append("")

        conflicts = check_conflicts(rules)

        lines.append("### 2. 策略冲突检测")

        if conflicts:
            lines.append("检测到策略冲突，系统拒绝下发策略。")
            for c in conflicts:
                lines.append(
                    f"- `{c['src']} -> {c['dst']}` 同时存在 `{c['actions']}`"
                )

            lines.append("")
            lines.append("### 3. 执行结论")
            lines.append("为了避免错误流表影响网络运行，本次策略未下发。")
            return "\n".join(lines)

        lines.append("未检测到冲突，可以继续下发策略。\n")

        apply_results = []
        for rule in rules:
            result = apply_rule(
                rule["src"],
                rule["dst"],
                rule["action"]
            )
            apply_results.append(result)

        lines.append("### 3. SDN / Ryu 策略下发结果")
        for r in apply_results:
            lines.append(
                f"- `{r['src']} -> {r['dst']} : {r['action']}`，status = `{r['status']}`"
            )
        lines.append("")

        lines.append("### 4. Mininet 闭环验证命令")
        for rule in rules:
            if rule["action"] == "deny":
                cmd = f"{rule['src']} ping -c 3 {rule['dst']}"
                lines.append(f"- `{cmd}`，预期：失败")
            elif rule["action"] == "allow":
                cmd = f"{rule['src']} ping -c 3 {rule['dst']}"
                lines.append(f"- `{cmd}`，预期：成功")

        lines.append("")
        lines.append("### 5. 执行报告")
        lines.append(
            "系统已完成自然语言意图解析、策略生成、冲突检测、Ryu/OpenFlow 流表下发，并生成验证命令。"
        )

        return "\n".join(lines)

    except Exception as e:
        return f"系统执行失败：{str(e)}"


def submit_intent(text):
    st.session_state.messages.append({
        "role": "user",
        "content": text
    })

    with st.spinner("智能体正在解析并执行网络意图..."):
        reply = execute_intent(text)

    st.session_state.messages.append({
        "role": "assistant",
        "content": reply
    })
