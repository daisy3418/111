# 多会话状态：模拟豆包/ChatGPT 左侧历史对话
# =========================
if "chats" not in st.session_state:
    st.session_state.chats = {
        "chat_1": {
            "title": "新对话",
            "messages": []
        }
    }

if "current_chat_id" not in st.session_state:
    st.session_state.current_chat_id = "chat_1"

if "chat_counter" not in st.session_state:
    st.session_state.chat_counter = 1


def get_current_chat():
    return st.session_state.chats[st.session_state.current_chat_id]


def new_chat():
    st.session_state.chat_counter += 1
    chat_id = "chat_" + str(st.session_state.chat_counter)
    st.session_state.chats[chat_id] = {
        "title": "新对话",
        "messages": []
    }
    st.session_state.current_chat_id = chat_id


def set_chat_title_if_needed(user_text):
    chat = get_current_chat()
    if chat["title"] == "新对话":
        title = user_text.strip()
        if len(title) > 16:
            title = title[:16] + "..."
        chat["title"] = title


def add_message(role, content):
    chat = get_current_chat()
    chat["messages"].append({
        "role": role,
        "content": content
    })

# 把 network_agent 的结构化结果转成前端展示文本
# =========================
def format_agent_result(result):
    lines = []

    task_type = result.get("task_type", "unknown")
    summary = result.get("summary", "")

    lines.append("### 🧠 智能体任务识别")
    lines.append("- 任务类型：`{}`".format(task_type))
    lines.append("- 执行摘要：{}".format(summary))
    lines.append("")

    plan = result.get("plan", [])
    if plan:
        lines.append("### 1. 配置执行计划")
        for i, step in enumerate(plan, start=1):
            lines.append("{}. {}".format(i, step))
        lines.append("")

    rules = result.get("rules", [])
    if rules:
        lines.append("### 2. 结构化网络配置策略")
        for rule in rules:
            lines.append(
                "- `{}` → `{}` ：`{}`".format(
                    rule["src"],
                    rule["dst"],
                    rule["action"]
                )
            )
        lines.append("")

    conflicts = result.get("conflicts", [])
    lines.append("### 3. 策略冲突检测")

    if conflicts:
        lines.append("检测到策略冲突，系统拒绝下发配置。")
        for c in conflicts:
            lines.append(
                "- `{}` → `{}` 存在冲突动作：`{}`".format(
                    c["src"],
                    c["dst"],
                    c["actions"]
                )
            )
        lines.append("")
        lines.append("### 4. 智能体反馈")
        lines.append(summary)
        return "\n".join(lines)

    if task_type == "configure":
        lines.append("未检测到策略冲突。")
    else:
        lines.append("当前任务无需进行策略冲突检测。")
    lines.append("")

    apply_results = result.get("apply_results", [])
    if apply_results:
        lines.append("### 4. Ryu / OpenFlow 配置下发结果")
        for r in apply_results:
            lines.append(
                "- `{}` → `{}` ：`{}`，dpid=`{}`，status=`{}`".format(
                    r["src"],
                    r["dst"],
                    r["action"],
                    r.get("dpid", "-"),
                    r.get("status", "-")
                )
            )
        lines.append("")

    verification = result.get("verification", [])
    if verification:
        lines.append("### 5. Mininet 闭环验证命令")
        for item in verification:
            lines.append(
                "- `{}`，预期：`{}`".format(
                    item["cmd"],
                    item["expected"]
                )
            )
        lines.append("")

    lines.append("### 6. 智能体反馈")
    lines.append(summary)

    return "\n".join(lines)


def submit_intent(text):
    set_chat_title_if_needed(text)

    add_message("user", text)

    with st.spinner("网络配置智能体正在理解意图、规划策略并执行配置..."):
        result = run_network_agent(text)
        reply = format_agent_result(result)

    add_message("assistant", reply)
