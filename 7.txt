from agent.network_agent import run_network_agent


def print_result(result):
    print("\n========== NetIntent-Shield Agent Report ==========")

    print("\n[1] Task Type")
    print(result.get("task_type", "unknown"))

    print("\n[2] Summary")
    print(result.get("summary", ""))

    plan = result.get("plan", [])
    print("\n[3] Configuration Plan")
    if plan:
        for i, step in enumerate(plan, start=1):
            print(str(i) + ". " + step)
    else:
        print("No plan generated.")

    rules = result.get("rules", [])
    print("\n[4] Structured Rules")
    if rules:
        for r in rules:
            print(r)
    else:
        print("No rules generated.")

    conflicts = result.get("conflicts", [])
    print("\n[5] Conflict Check")
    if conflicts:
        print("Conflict detected:")
        for c in conflicts:
            print(c)
        print("Policy will not be applied.")
    else:
        print("No conflict detected.")

    apply_results = result.get("apply_results", [])
    print("\n[6] Apply Results")
    if apply_results:
        for r in apply_results:
            print(r)
    else:
        print("No policy applied.")

    verification = result.get("verification", [])
    print("\n[7] Verification Commands")
    if verification:
        for item in verification:
            print(item["cmd"] + "    expected: " + item["expected"])
    else:
        print("No verification command generated.")

    print("\n===================================================")


def main():
    print("NetIntent-Shield Network Configuration Agent")
    print("Input natural language network configuration intent.")
    print("Example: 配置服务器区保护策略")
    print("Example: 按照最小权限原则配置当前内网")
    print("Example: 保护管理安全区，禁止异常主机访问管理员主机和监控审计主机")
    print("Type exit to quit.\n")

    while True:
        user_input = input("Intent> ").strip()

        if user_input.lower() in ["exit", "quit"]:
            break

        if not user_input:
            continue

        result = run_network_agent(user_input)
        print_result(result)


if __name__ == "__main__":
    main()
