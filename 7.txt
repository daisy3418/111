from agent.llm_agent import agent_parse_intent
from agent.simple_parser import parse_intent
from sdn.conflict_checker import check_conflicts
from sdn.policy_apply import apply_rule


def classify_task(user_text):
    text = user_text.lower()

    rollback_words = ["回滚", "撤销", "恢复", "清空", "还原"]
    verify_words = ["验证", "测试", "检查是否生效", "怎么验证", "如何验证"]

    configure_words = [
        "配置", "策略", "访问", "禁止", "允许", "隔离", "保护",
        "最小权限", "服务器区", "管理安全区", "数据库",
        "web", "普通用户", "正常用户", "异常主机", "攻击主机",
        "管理员", "监控", "h1", "h2", "h3", "h4", "h5", "h6",
        "allow", "deny", "block", "isolate"
    ]

    for w in rollback_words:
        if w in text:
            return "rollback"

    for w in verify_words:
        if w in text:
            return "verify"

    for w in configure_words:
        if w in text:
            return "configure"

    return "invalid"


def build_plan_from_rules(rules):
    plan = []

    for rule in rules:
        src = rule["src"]
        dst = rule["dst"]
        action = rule["action"]

        if action == "deny":
            plan.append("阻断 {} 到 {} 的访问流量".format(src, dst))
        elif action == "allow":
            plan.append("允许 {} 到 {} 的访问关系".format(src, dst))
        else:
            plan.append("处理 {} 到 {} 的未知动作 {}".format(src, dst, action))

    return plan


def build_verification_from_rules(rules):
    verification = []

    for rule in rules:
        src = rule["src"]
        dst = rule["dst"]
        action = rule["action"]

        if action == "deny":
            verification.append({
                "cmd": "{} ping -c 3 {}".format(src, dst),
                "expected": "fail"
            })
        elif action == "allow":
            verification.append({
                "cmd": "{} ping -c 3 {}".format(src, dst),
                "expected": "success"
            })

    return verification


def run_network_agent(user_text):
    task_type = classify_task(user_text)

    if task_type == "invalid":
        return {
            "task_type": "invalid",
            "summary": "我没有识别到有效的网络配置意图。请描述你希望配置的访问关系、安全域策略、隔离目标或验证目标。",
            "plan": [],
            "rules": [],
            "conflicts": [],
            "apply_results": [],
            "verification": []
        }

    if task_type == "verify":
        return {
            "task_type": "verify",
            "summary": "这是一个验证类请求。请结合最近一次下发的策略，在 Mininet 中执行对应 ping 命令进行验证。当前版本会在每次配置后自动生成验证命令。",
            "plan": ["根据最近一次策略生成验证命令", "在 Mininet 中执行 ping 测试", "对比预期结果"],
            "rules": [],
            "conflicts": [],
            "apply_results": [],
            "verification": []
        }

    if task_type == "rollback":
        return {
            "task_type": "rollback",
            "summary": "这是一个回滚类请求。当前版本暂未真正清除流表，下一步将接入 Ryu 流表清理接口实现配置回滚。",
            "plan": ["识别回滚请求", "准备清理已下发策略", "恢复默认网络连通性"],
            "rules": [],
            "conflicts": [],
            "apply_results": [],
            "verification": []
        }

    try:
        try:
            rules = agent_parse_intent(user_text)
            parser_name = "LangChain 网络配置意图智能体"
        except Exception:
            rules = parse_intent(user_text)
            parser_name = "本地规则解析器"

        plan = build_plan_from_rules(rules)
        conflicts = check_conflicts(rules)

        apply_results = []

        if not conflicts:
            for rule in rules:
                result = apply_rule(
                    rule["src"],
                    rule["dst"],
                    rule["action"]
                )
                apply_results.append(result)

        verification = build_verification_from_rules(rules)

        if conflicts:
            summary = "智能体已生成配置策略，但检测到策略冲突，因此未下发到网络。"
        else:
            summary = "智能体已完成配置意图解析、策略编排、冲突检测和 Ryu/OpenFlow 下发。"

        return {
            "task_type": "configure",
            "parser": parser_name,
            "summary": summary,
            "plan": plan,
            "rules": rules,
            "conflicts": conflicts,
            "apply_results": apply_results,
            "verification": verification
        }

    except Exception as e:
        return {
            "task_type": "error",
            "summary": "系统执行失败：{}".format(str(e)),
            "plan": [],
            "rules": [],
            "conflicts": [],
            "apply_results": [],
            "verification": []
        }
