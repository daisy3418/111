def execute_intent(user_text):
    lines = []

    try:
        try:
            rules = agent_parse_intent(user_text)
            parser_name = "LangChain 网络配置意图智能体"
        except Exception as llm_error:
            rules = parse_intent(user_text)
            parser_name = "本地规则解析器"

        lines.append("### 🤖 解析方式")
        lines.append(parser_name)
        lines.append("")

        lines.append("### 1. 网络配置意图解析结果")
        lines.append("系统已将自然语言配置意图转换为结构化策略规则：")
        lines.append("")

        for rule in rules:
            lines.append(
                "- `{}` → `{}` ：`{}`".format(
                    rule["src"],
                    rule["dst"],
                    rule["action"]
                )
            )

        lines.append("")

        conflicts = check_conflicts(rules)

        lines.append("### 2. 策略冲突检测")

        if conflicts:
            lines.append("检测到策略冲突，系统拒绝下发配置。")
            lines.append("")

            for c in conflicts:
                lines.append(
                    "- `{}` → `{}` 同时存在 `{}`".format(
                        c["src"],
                        c["dst"],
                        c["actions"]
                    )
                )

            lines.append("")
            lines.append("### 3. 执行结论")
            lines.append(
                "为了避免错误 OpenFlow 流表影响网络运行，本次配置未下发到 Ryu 控制器。"
            )

            return "\n".join(lines)

        lines.append("未检测到策略冲突，可以继续下发网络配置。")
        lines.append("")

        apply_results = []

        for rule in rules:
            result = apply_rule(
                rule["src"],
                rule["dst"],
                rule["action"]
            )
            apply_results.append(result)

        lines.append("### 3. SDN / Ryu 配置下发结果")

        for r in apply_results:
            lines.append(
                "- `{}` → `{}` ：`{}`，dpid=`{}`，status=`{}`".format(
                    r["src"],
                    r["dst"],
                    r["action"],
                    r.get("dpid", "-"),
                    r["status"]
                )
            )

        lines.append("")

        lines.append("### 4. Mininet 闭环验证命令")

        for rule in rules:
            if rule["action"] == "deny":
                cmd = "{} ping -c 3 {}".format(rule["src"], rule["dst"])
                lines.append("- `{}`，预期：失败".format(cmd))
            elif rule["action"] == "allow":
                cmd = "{} ping -c 3 {}".format(rule["src"], rule["dst"])
                lines.append("- `{}`，预期：成功".format(cmd))

        lines.append("")
        lines.append("### 5. 执行报告")
        lines.append(
            "系统已完成：自然语言配置意图解析 → 策略生成 → 冲突检测 → "
            "Ryu/OpenFlow 配置下发 → Mininet 验证命令生成。"
        )

        return "\n".join(lines)

    except Exception as e:
        return "系统执行失败：{}".format(str(e))


def submit_intent(text):
    st.session_state.messages.append({
        "role": "user",
        "content": text
    })

    with st.spinner("智能体正在解析并执行网络配置意图..."):
        reply = execute_intent(text)

    st.session_state.messages.append({
        "role": "assistant",
        "content": reply
    })
