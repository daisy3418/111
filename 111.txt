def run_intent(user_text):
    result_lines = []

    try:
        try:
            rules = agent_parse_intent(user_text)
            parser_name = "LangChain Intent Agent"
        except Exception as llm_error:
            rules = parse_intent(user_text)
            parser_name = "本地规则解析器"

        result_lines.append("### 解析方式")
        result_lines.append(parser_name)
        result_lines.append("")

        result_lines.append("### AI 解析出的策略规则")
        for rule in rules:
            result_lines.append(
                "- {} → {} ：{}".format(
                    rule["src"],
                    rule["dst"],
                    rule["action"]
                )
            )
        result_lines.append("")

        conflicts = check_conflicts(rules)

        result_lines.append("### 策略冲突检测")

        if conflicts:
            result_lines.append("检测到策略冲突，系统拒绝下发策略。")
            result_lines.append("")

            for c in conflicts:
                result_lines.append(
                    "- {} → {} 同时存在 {}".format(
                        c["src"],
                        c["dst"],
                        c["actions"]
                    )
                )

            result_lines.append("")
            result_lines.append("### 执行结论")
            result_lines.append("为了避免错误流表影响网络运行，本次策略未下发。")

            return "\n".join(result_lines)

        result_lines.append("未检测到冲突，可以下发策略。")
        result_lines.append("")

        apply_results = []

        for rule in rules:
            r = apply_rule(
                rule["src"],
                rule["dst"],
                rule["action"]
            )
            apply_results.append(r)

        result_lines.append("### SDN 策略下发结果")

        for r in apply_results:
            result_lines.append(
                "- {} → {} ：{}，status={}".format(
                    r["src"],
                    r["dst"],
                    r["action"],
                    r["status"]
                )
            )

        result_lines.append("")
        result_lines.append("### Mininet 闭环验证命令")

        for rule in rules:
            if rule["action"] == "deny":
                result_lines.append(
                    "- `{}`，预期：失败".format(
                        rule["src"] + " ping -c 3 " + rule["dst"]
                    )
                )
            elif rule["action"] == "allow":
                result_lines.append(
                    "- `{}`，预期：成功".format(
                        rule["src"] + " ping -c 3 " + rule["dst"]
                    )
                )

        result_lines.append("")
        result_lines.append("### 执行报告")
        result_lines.append(
            "系统已完成自然语言意图解析、结构化策略生成、冲突检测、Ryu/OpenFlow 流表下发，并生成验证命令。"
        )

        return "\n".join(result_lines)

    except Exception as e:
        return "系统执行失败：{}".format(str(e))
