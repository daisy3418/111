def build_agent_reply(rules, conflicts, apply_results):
    lines = []

    lines.append("我已完成网络意图解析。")
    lines.append("")
    lines.append("**解析出的策略规则：**")
    for rule in rules:
        lines.append(f"- {rule['src']} → {rule['dst']}：{rule['action']}")

    lines.append("")

    if conflicts:
        lines.append("**冲突检测结果：发现冲突。**")
        for c in conflicts:
            lines.append(f"- {c['src']} → {c['dst']} 同时存在 {c['actions']}")
        lines.append("")
        lines.append("为避免错误策略影响网络运行，本次策略未下发。")
        return "\n".join(lines)

    lines.append("**冲突检测结果：未发现冲突。**")
    lines.append("")
    lines.append("**策略下发结果：**")

    for r in apply_results:
        lines.append(f"- {r['src']} → {r['dst']}：{r['action']}，status={r['status']}")

    lines.append("")
    lines.append("**建议在 Mininet 中执行以下验证命令：**")

    for rule in rules:
        if rule["action"] == "deny":
            lines.append(f"- {rule['src']} ping -c 3 {rule['dst']}，预期失败")
        elif rule["action"] == "allow":
            lines.append(f"- {rule['src']} ping -c 3 {rule['dst']}，预期成功")

    return "\n".join(lines)

            rules = agent_parse_intent(user_input)
            parser_name = "LangChain Intent Agent"
        except Exception as llm_error:
            rules = parse_intent(user_input)
            parser_name = "本地规则解析器"

        conflicts = check_conflicts(rules)
        apply_results = []

        if not conflicts:
            for rule in rules:
                result = apply_rule(
                    rule["src"],
                    rule["dst"],
                    rule["action"]
                )
                apply_results.append(result)

        reply = build_agent_reply(rules, conflicts, apply_results)
        reply = f"解析方式：**{parser_name}**\n\n" + reply

        st.session_state.history.append({
            "role": "agent",
            "content": reply
        })

        st.rerun()

    except Exception as e:
        st.session_state.history.append({
            "role": "error",
            "content": str(e)
        })
        st.rerun()
